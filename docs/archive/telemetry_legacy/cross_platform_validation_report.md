#### 3. Environment Detection
```bash
# Automatic platform detection
detect_platform() {
  case "$(uname -s)" in
    CYGWIN*|MINGW*|MSYS*)
      echo "windows"
      ;;
    Darwin)
      echo "macos"
      ;;
    Linux)
      echo "linux"
      ;;
    *)
      echo "unknown"
      ;;
  esac
}
```
